119 static double grad(
120 int hash
121 , double x
122 , double y
123 , double z
124 ) {
125 int h = hash & 15; // Take the hashed value and take the first 4 bits of it (15 == 0b1111)
126 double u = h < 8 /* 0b1000 */ ? x : y; // If the most significant bit (MSB) of the hash is 0 then set u = x. Otherwise y.
127
128 double v; // In Ken Perlin's original implementation this was another conditional operator (?:). I
129 // expanded it for readability.
130
131 if (h < 4 /* 0b0100 */) // If the first and second significant bits are 0 set v = y
132 v = y;
133 else if (h == 12 /* 0b1100 */ || h == 14 /* 0b1110*/)// If the first and second significant bits are 1 set v = x
134 v = x;
135 else // If the first and second significant bits are not equal (0/1, 1/0) set v = z
136 v = z;
137
138 return ((h & 1) == 0 ? u : -u) + ((h & 2) == 0 ? v : -v); // Use the last 2 bits to decide if u and v are positive or negative. Then return their addition.
139 }
